99  void eval_right_hand_side(Eigen::Ref<const Eigen::VectorXd> y, ScalarType t, Eigen::Ref<Eigen::VectorXd> dydt) const
100  {
101  dydt.setZero();
102 
103  ScalarType C = 0;
104  ScalarType I = 0;
105  ScalarType dummy = 0;
106 
107  // Calculate sum of all subcompartments for InfectedNoSymptoms.
108  C = y.segment(LctState::template get_first_index<InfectionState::InfectedNoSymptoms>(),
109  LctState::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>())
110  .sum();
111  // Calculate sum of all subcompartments for InfectedSymptoms.
112  I = y.segment(LctState::template get_first_index<InfectionState::InfectedSymptoms>(),
113  LctState::template get_num_subcompartments<InfectionState::InfectedSymptoms>())
114  .sum();
115 
116  // S'
117  ScalarType season_val =
118  1 + parameters.get<Seasonality>() *
119  sin(3.141592653589793 * (std::fmod((parameters.get<StartDay>() + t), 365.0) / 182.5 + 0.5));
120  dydt[0] = -y[0] / (m_N0 - y[LctState::template get_first_index<InfectionState::Dead>()]) * season_val *
121  parameters.get<TransmissionProbabilityOnContact>() *
122  parameters.get<ContactPatterns>().get_cont_freq_mat().get_matrix_at(t)(0, 0) *
123  (parameters.get<RelativeTransmissionNoSymptoms>() * C +
124  parameters.get<RiskOfInfectionFromSymptomatic>() * I);
125 
126  // E'
127  dydt[1] = -dydt[0];
128  for (Eigen::Index i = 0; i < LctState::template get_num_subcompartments<InfectionState::Exposed>(); i++) {
129  // Dummy stores the value of the flow from dydt[1 + i] to dydt[2 + i].
130  // 1+i is always the index of a (sub-)compartment of E and 2+i can also be the index of the first (sub-)compartment of C.
131  dummy = LctState::template get_num_subcompartments<InfectionState::Exposed>() *
132  (1 / parameters.get<TimeExposed>()) * y[1 + i];
133  // Subtract flow from dydt[1 + i] and add to dydt[2 + i].
134  dydt[1 + i] = dydt[1 + i] - dummy;
135  dydt[2 + i] = dummy;
136  }
137 
138  // C'
139  for (Eigen::Index i = 0; i < LctState::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>();
140  i++) {
141  dummy = LctState::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>() *
142  (1 / parameters.get<TimeInfectedNoSymptoms>()) *
143  y[LctState::template get_first_index<InfectionState::InfectedNoSymptoms>() + i];
144  dydt[LctState::template get_first_index<InfectionState::InfectedNoSymptoms>() + i] =
145  dydt[LctState::template get_first_index<InfectionState::InfectedNoSymptoms>() + i] - dummy;
146  dydt[LctState::template get_first_index<InfectionState::InfectedNoSymptoms>() + i + 1] = dummy;
147  }
148 
149  // I'
150  // Flow from last (sub-) compartment of C must be split between I_1 and R.
151  dydt[LctState::template get_first_index<InfectionState::Recovered>()] =
152  dydt[LctState::template get_first_index<InfectionState::InfectedSymptoms>()] *
153  parameters.get<RecoveredPerInfectedNoSymptoms>();
154  dydt[LctState::template get_first_index<InfectionState::InfectedSymptoms>()] =
155  dydt[LctState::template get_first_index<InfectionState::InfectedSymptoms>()] *
156  (1 - parameters.get<RecoveredPerInfectedNoSymptoms>());
157 
158  for (Eigen::Index i = 0; i < LctState::template get_num_subcompartments<InfectionState::InfectedSymptoms>();
159  i++) {
160  dummy = LctState::template get_num_subcompartments<InfectionState::InfectedSymptoms>() *
161  (1 / parameters.get<TimeInfectedSymptoms>()) *
162  y[LctState::template get_first_index<InfectionState::InfectedSymptoms>() + i];
163  dydt[LctState::template get_first_index<InfectionState::InfectedSymptoms>() + i] =
164  dydt[LctState::template get_first_index<InfectionState::InfectedSymptoms>() + i] - dummy;
165  dydt[LctState::template get_first_index<InfectionState::InfectedSymptoms>() + i + 1] = dummy;
166  }
167 
168  // H'
169  dydt[LctState::template get_first_index<InfectionState::Recovered>()] =
170  dydt[LctState::template get_first_index<InfectionState::Recovered>()] +
171  dydt[LctState::template get_first_index<InfectionState::InfectedSevere>()] *
172  (1 - parameters.get<SeverePerInfectedSymptoms>());
173  dydt[LctState::template get_first_index<InfectionState::InfectedSevere>()] =
174  dydt[LctState::template get_first_index<InfectionState::InfectedSevere>()] *
175  parameters.get<SeverePerInfectedSymptoms>();
176  for (Eigen::Index i = 0; i < LctState::template get_num_subcompartments<InfectionState::InfectedSevere>();
177  i++) {
178  dummy = LctState::template get_num_subcompartments<InfectionState::InfectedSevere>() *
179  (1 / parameters.get<TimeInfectedSevere>()) *
180  y[LctState::template get_first_index<InfectionState::InfectedSevere>() + i];
181  dydt[LctState::template get_first_index<InfectionState::InfectedSevere>() + i] =
182  dydt[LctState::template get_first_index<InfectionState::InfectedSevere>() + i] - dummy;
183  dydt[LctState::template get_first_index<InfectionState::InfectedSevere>() + i + 1] = dummy;
184  }
185 
186  // U'
187  dydt[LctState::template get_first_index<InfectionState::Recovered>()] =
188  dydt[LctState::template get_first_index<InfectionState::Recovered>()] +
189  dydt[LctState::template get_first_index<InfectionState::InfectedCritical>()] *
190  (1 - parameters.get<CriticalPerSevere>());
191  dydt[LctState::template get_first_index<InfectionState::InfectedCritical>()] =
192  dydt[LctState::template get_first_index<InfectionState::InfectedCritical>()] *
193  parameters.get<CriticalPerSevere>();
194  for (Eigen::Index i = 0; i < LctState::template get_num_subcompartments<InfectionState::InfectedCritical>() - 1;
195  i++) {
196  dummy = LctState::template get_num_subcompartments<InfectionState::InfectedCritical>() *
197  (1 / parameters.get<TimeInfectedCritical>()) *
198  y[LctState::template get_first_index<InfectionState::InfectedCritical>() + i];
199  dydt[LctState::template get_first_index<InfectionState::InfectedCritical>() + i] =
200  dydt[LctState::template get_first_index<InfectionState::InfectedCritical>() + i] - dummy;
201  dydt[LctState::template get_first_index<InfectionState::InfectedCritical>() + i + 1] = dummy;
202  }
203  // Last flow from U has to be divided between R and D.
204  // Must be calculated separately in order not to overwrite the already calculated values ​​for R.
205  dummy = LctState::template get_num_subcompartments<InfectionState::InfectedCritical>() *
206  (1 / parameters.get<TimeInfectedCritical>()) *
207  y[LctState::template get_first_index<InfectionState::Recovered>() - 1];
208  dydt[LctState::template get_first_index<InfectionState::Recovered>() - 1] =
209  dydt[LctState::template get_first_index<InfectionState::Recovered>() - 1] - dummy;
210  dydt[LctState::template get_first_index<InfectionState::Recovered>()] =
211  dydt[LctState::template get_first_index<InfectionState::Recovered>()] +
212  (1 - parameters.get<DeathsPerCritical>()) * dummy;
213  dydt[LctState::template get_first_index<InfectionState::Dead>()] = parameters.get<DeathsPerCritical>() * dummy;
214  }

226  TimeSeries<ScalarType> calculate_populations(const TimeSeries<ScalarType>& result) const
227  {
228  if (!(LctState::Count == result.get_num_elements())) {
229  log_error("Result does not match infectionState of the Model.");
230  TimeSeries<ScalarType> populations((int)InfectionState::Count);
231  Eigen::VectorXd wrong_size = Eigen::VectorXd::Constant((int)InfectionState::Count, -1);
232  populations.add_time_point(-1, wrong_size);
233  return populations;
234  }
235  TimeSeries<ScalarType> populations((int)InfectionState::Count);
236  Eigen::VectorXd dummy((int)InfectionState::Count);
237  for (Eigen::Index i = 0; i < result.get_num_time_points(); ++i) {
238  // Use segment of vector of the result with subcompartments of InfectionState with index j and sum up values of subcompartments.
239  dummy[(int)InfectionState::Susceptible] = result[i][0];
240  dummy[(int)InfectionState::Exposed] =
241  result[i]
242  .segment(LctState::template get_first_index<InfectionState::Exposed>(),
243  LctState::template get_num_subcompartments<InfectionState::Exposed>())
244  .sum();
245  dummy[(int)InfectionState::InfectedNoSymptoms] =
246  result[i]
247  .segment(LctState::template get_first_index<InfectionState::InfectedNoSymptoms>(),
248  LctState::template get_num_subcompartments<InfectionState::InfectedNoSymptoms>())
249  .sum();
250  dummy[(int)InfectionState::InfectedSymptoms] =
251  result[i]
252  .segment(LctState::template get_first_index<InfectionState::InfectedSymptoms>(),
253  LctState::template get_num_subcompartments<InfectionState::InfectedSymptoms>())
254  .sum();
255  dummy[(int)InfectionState::InfectedSevere] =
256  result[i]
257  .segment(LctState::template get_first_index<InfectionState::InfectedSevere>(),
258  LctState::template get_num_subcompartments<InfectionState::InfectedSevere>())
259  .sum();
260  dummy[(int)InfectionState::InfectedCritical] =
261  result[i]
262  .segment(LctState::template get_first_index<InfectionState::InfectedCritical>(),
263  LctState::template get_num_subcompartments<InfectionState::InfectedCritical>())
264  .sum();
265  dummy[(int)InfectionState::Recovered] =
266  result[i][LctState::template get_first_index<InfectionState::Recovered>()];
267  dummy[(int)InfectionState::Dead] = result[i][LctState::template get_first_index<InfectionState::Dead>()];
268 
269  populations.add_time_point(result.get_time(i), dummy);
270  }
271 
272  return populations;
273  }
