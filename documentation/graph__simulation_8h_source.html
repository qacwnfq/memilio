59  void advance(double t_max = 1.0)
60  {
61  auto dt = m_dt;
62  while (m_t < t_max) {
63  if (m_t + dt > t_max) {
64  dt = t_max - m_t;
65  }
66 
67  for (auto& n : m_graph.nodes()) {
68  m_node_func(m_t, dt, n.property);
69  }
70 
71  m_t += dt;
72 
73  for (auto& e : m_graph.edges()) {
74  m_edge_func(m_t, dt, e.property, m_graph.nodes()[e.start_node_idx].property,
75  m_graph.nodes()[e.end_node_idx].property);
76  }
77  }
