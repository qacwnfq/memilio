53  void get_flows(Eigen::Ref<const Eigen::VectorXd> pop, Eigen::Ref<const Eigen::VectorXd> y, double t,
54  Eigen::Ref<Eigen::VectorXd> flows) const
55  {
56  auto& params = this->parameters;
57  double coeffStoI = params.get<ContactPatterns>().get_matrix_at(t)(0, 0) *
58  params.get<TransmissionProbabilityOnContact>() / populations.get_total();
59 
60  double si = mio::DistributionAdapter<std::normal_distribution<double>>::get_instance()(rng, 0.0, 1.0);
61  double ir = mio::DistributionAdapter<std::normal_distribution<double>>::get_instance()(rng, 0.0, 1.0);
62  double rs = mio::DistributionAdapter<std::normal_distribution<double>>::get_instance()(rng, 0.0, 1.0);
63 
64  const double inv_sqrt_dt = 1 / sqrt(step_size);
65 
66  // Assuming that no person can change its InfectionState twice in a single time step,
67  // take the minimum of the calculated flow and the source compartment, to ensure that
68  // no compartment attains negative values.
69 
70  flows[get_flat_flow_index<InfectionState::Susceptible, InfectionState::Infected>()] = std::min(
71  coeffStoI * y[(size_t)InfectionState::Susceptible] * pop[(size_t)InfectionState::Infected] +
72  sqrt(coeffStoI * y[(size_t)InfectionState::Susceptible] * pop[(size_t)InfectionState::Infected]) *
73  inv_sqrt_dt * si,
74  y[(size_t)InfectionState::Susceptible] / step_size);
75 
76  flows[get_flat_flow_index<InfectionState::Infected, InfectionState::Recovered>()] = std::min(
77  (1.0 / params.get<TimeInfected>()) * y[(size_t)InfectionState::Infected] +
78  sqrt((1.0 / params.get<TimeInfected>()) * y[(size_t)InfectionState::Infected]) * inv_sqrt_dt * ir,
79  y[(size_t)InfectionState::Infected] / step_size);
80 
81  flows[get_flat_flow_index<InfectionState::Recovered, InfectionState::Susceptible>()] = std::min(
82  (1.0 / params.get<TimeImmune>()) * y[(size_t)InfectionState::Recovered] +
83  sqrt((1.0 / params.get<TimeImmune>()) * y[(size_t)InfectionState::Recovered]) * inv_sqrt_dt * rs,
84  y[(size_t)InfectionState::Recovered] / step_size);
85  }
86 
87  ScalarType step_size = 0.1;
88  mutable RandomNumberGenerator rng;
89

91 };
