616  bool check_constraints() const
617  {
618  for (auto i = AgeGroup(0); i < AgeGroup(m_num_groups); ++i) {
619 
620  if (this->get<IncubationPeriod>()[{VirusVariant::Wildtype, i}] < 0) {
621  log_error("Constraint check: Parameter IncubationPeriod of age group {:.0f} smaller than {:.4f}",
622  (size_t)i, 0);
623  return true;
624  }
625 
626  if (this->get<InfectedNoSymptomsToSymptoms>()[{VirusVariant::Wildtype, i}] < 0.0) {
627  log_error("Constraint check: Parameter InfectedNoSymptomsToSymptoms of age group {:.0f} smaller "
628  "than {:d}",
629  (size_t)i, 0);
630  return true;
631  }
632 
633  if (this->get<InfectedNoSymptomsToRecovered>()[{VirusVariant::Wildtype, i}] < 0.0) {
634  log_error("Constraint check: Parameter InfectedNoSymptomsToRecovered of age group {:.0f} smaller "
635  "than {:d}",
636  (size_t)i, 0);
637  return true;
638  }
639 
640  if (this->get<InfectedSymptomsToRecovered>()[{VirusVariant::Wildtype, i}] < 0.0) {
641  log_error(
642  "Constraint check: Parameter InfectedSymptomsToRecovered of age group {:.0f} smaller than {:d}",
643  (size_t)i, 0);
644  return true;
645  }
646 
647  if (this->get<InfectedSymptomsToSevere>()[{VirusVariant::Wildtype, i}] < 0.0) {
648  log_error("Constraint check: Parameter InfectedSymptomsToSevere of age group {:.0f} smaller than {:d}",
649  (size_t)i, 0);
650  return true;
651  }
652 
653  if (this->get<SevereToCritical>()[{VirusVariant::Wildtype, i}] < 0.0) {
654  log_error("Constraint check: Parameter SevereToCritical of age group {:.0f} smaller than {:d}",
655  (size_t)i, 0);
656  return true;
657  }
658 
659  if (this->get<SevereToRecovered>()[{VirusVariant::Wildtype, i}] < 0.0) {
660  log_error("Constraint check: Parameter SevereToRecovered of age group {:.0f} smaller than {:d}",
661  (size_t)i, 0);
662  return true;
663  }
664 
665  if (this->get<CriticalToDead>()[{VirusVariant::Wildtype, i}] < 0.0) {
666  log_error("Constraint check: Parameter CriticalToDead of age group {:.0f} smaller than {:d}", (size_t)i,
667  0);
668  return true;
669  }
670 
671  if (this->get<CriticalToRecovered>()[{VirusVariant::Wildtype, i}] < 0.0) {
672  log_error("Constraint check: Parameter CriticalToRecovered of age group {:.0f} smaller than {:d}",
673  (size_t)i, 0);
674  return true;
675  }
676 
677  if (this->get<RecoveredToSusceptible>()[{VirusVariant::Wildtype, i}] < 0.0) {
678  log_error("Constraint check: Parameter RecoveredToSusceptible of age group {:.0f} smaller than {:d}",
679  (size_t)i, 0);
680  return true;
681  }
682 
683  if (this->get<DetectInfection>()[{VirusVariant::Wildtype, i}] < 0.0 ||
684  this->get<DetectInfection>()[{VirusVariant::Wildtype, i}] > 1.0) {
685  log_error("Constraint check: Parameter DetectInfection of age group {:.0f} smaller than {:d} or "
686  "larger than {:d}",
687  (size_t)i, 0, 1);
688  return true;
689  }
690 
691  if (this->get<GotoWorkTimeMinimum>()[i].seconds() < 0.0 ||
692  this->get<GotoWorkTimeMinimum>()[i].seconds() > this->get<GotoWorkTimeMaximum>()[i].seconds()) {
693  log_error("Constraint check: Parameter GotoWorkTimeMinimum of age group {:.0f} smaller {:d} or "
694  "larger {:d}",
695  (size_t)i, 0, this->get<GotoWorkTimeMaximum>()[i].seconds());
696  return true;
697  }
698 
699  if (this->get<GotoWorkTimeMaximum>()[i].seconds() < this->get<GotoWorkTimeMinimum>()[i].seconds() ||
700  this->get<GotoWorkTimeMaximum>()[i] > days(1)) {
701  log_error("Constraint check: Parameter GotoWorkTimeMaximum of age group {:.0f} smaller {:d} or larger "
702  "than one day time span",
703  (size_t)i, this->get<GotoWorkTimeMinimum>()[i].seconds());
704  return true;
705  }
706 
707  if (this->get<GotoSchoolTimeMinimum>()[i].seconds() < 0.0 ||
708  this->get<GotoSchoolTimeMinimum>()[i].seconds() > this->get<GotoSchoolTimeMaximum>()[i].seconds()) {
709  log_error("Constraint check: Parameter GotoSchoolTimeMinimum of age group {:.0f} smaller {:d} or "
710  "larger {:d}",
711  (size_t)i, 0, this->get<GotoWorkTimeMaximum>()[i].seconds());
712  return true;
713  }
714 
715  if (this->get<GotoSchoolTimeMaximum>()[i].seconds() < this->get<GotoSchoolTimeMinimum>()[i].seconds() ||
716  this->get<GotoSchoolTimeMaximum>()[i] > days(1)) {
717  log_error("Constraint check: Parameter GotoWorkTimeMaximum of age group {:.0f} smaller {:d} or larger "
718  "than one day time span",
719  (size_t)i, this->get<GotoSchoolTimeMinimum>()[i].seconds());
720  return true;
721  }
722  }
723 
724  if (this->get<MaskProtection>()[MaskType::Community] < 0.0 ||
725  this->get<MaskProtection>()[MaskType::Community] > 1.0) {
726  log_error(
727  "Constraint check: Parameter MaskProtection for MaskType Community is smaller {:d} or larger {:d}", 0,
728  1);
729  return true;
730  }
731 
732  if (this->get<MaskProtection>()[MaskType::FFP2] < 0.0 || this->get<MaskProtection>()[MaskType::FFP2] > 1.0) {
733  log_error("Constraint check: Parameter MaskProtection for MaskType FFP2 is smaller {:d} or larger {:d}", 0,
734  1);
735  return true;
736  }
737 
738  if (this->get<MaskProtection>()[MaskType::Surgical] < 0.0 ||
739  this->get<MaskProtection>()[MaskType::Surgical] > 1.0) {
740  log_error("Constraint check: Parameter MaskProtection for MaskType Surgical smaller {:d} or larger {:d}", 0,
741  1);
742  return true;
743  }
744 
745  if (this->get<LockdownDate>().seconds() < 0.0) {
746  log_error("Constraint check: Parameter LockdownDate smaller {:d}", 0);
747  return true;
748  }
749 
750  return false;
751  }
