518  template <class Iter>
519  static void inclusive_exclusive_sum_rec(Iter b, Iter e, Matrix& sum)
520  {
521  if (b != e) {
522  sum = sum + std::get<Matrix>(*b) - (sum.array() * std::get<Matrix>(*b).array()).matrix();
523  inclusive_exclusive_sum_rec(++b, e, sum);
524  }
525  }
526  template <class Tuple>
527  static Matrix inclusive_exclusive_sum(const std::vector<Tuple>& v)
528  {
529  assert(!v.empty());
530  auto& m = std::get<Matrix>(v.front());
531  auto sum = m.eval();
532  inclusive_exclusive_sum_rec(v.begin() + 1, v.end(), sum);
533  return sum;
534  }

543 template <class D>
544 void Dampings<D>::update_cache()
545 {
546  using std::get;
547 
548  if (m_accumulated_dampings_cached.empty()) {
549  m_accumulated_dampings_cached.emplace_back(Matrix::Zero(m_shape.rows(), m_shape.cols()),
550  SimulationTime(std::numeric_limits<double>::lowest()));
551 
552  std::vector<std::tuple<std::reference_wrapper<const Matrix>, DampingLevel, DampingType>> active_by_type;
553  std::vector<std::tuple<Matrix, DampingLevel>> sum_by_level;
554  for (auto& damping : m_dampings) {
555  //update active damping
556  update_active_dampings(damping, active_by_type, sum_by_level);
557  auto combined_damping = inclusive_exclusive_sum(sum_by_level);
558  assert((combined_damping.array() <= 1).all() && (combined_damping.array() >= 0).all() &&
559  "unexpected error, accumulated damping out of range.");
560  if (floating_point_equal(double(get<SimulationTime>(damping)),
561  double(get<SimulationTime>(m_accumulated_dampings_cached.back())), 1e-15, 1e-15)) {
562  std::get<Matrix>(m_accumulated_dampings_cached.back()) = combined_damping;
563  }
564  else {
565  m_accumulated_dampings_cached.emplace_back(combined_damping, get<SimulationTime>(damping));
566  }
567  }
568 
569  m_accumulated_dampings_cached.emplace_back(get<Matrix>(m_accumulated_dampings_cached.back()),
570  SimulationTime(std::numeric_limits<double>::max()));
571  }
572 }

574 template <class D>
575 void Dampings<D>::add_(const value_type& damping)
576 {
577  assert(damping.get_shape() == m_shape && "Inconsistent matrix shape.");
578  insert_sorted_replace(m_dampings, damping, [](auto& tup1, auto& tup2) {
579  return std::make_tuple(tup1.get_time(), int(tup1.get_type()), int(tup1.get_level())) <
580  std::make_tuple(tup2.get_time(), int(tup2.get_type()), int(tup2.get_level()));
581  });
582  m_accumulated_dampings_cached.clear();
583  if (m_automatic_cache_update) {
584  update_cache();
585  }
586 }

588 template <class S>
589 void Dampings<S>::update_active_dampings(
590  const value_type& damping,
591  std::vector<std::tuple<std::reference_wrapper<const Matrix>, DampingLevel, DampingType>>& active_by_type,
592  std::vector<std::tuple<Matrix, DampingLevel>>& sum_by_level)
593 {
594  using std::get;
595 
596  const int MatrixIdx = 0;
597 
598  //find active with same type and level if existent
599  auto iter_active_same_type = std::find_if(active_by_type.begin(), active_by_type.end(), [&damping](auto& active) {
600  return get<DampingLevel>(active) == get<DampingLevel>(damping) &&
601  get<DampingType>(active) == get<DampingType>(damping);
602  });
603  if (iter_active_same_type != active_by_type.end()) {
604  //replace active of the same type and level
605  auto& active_same_type = *iter_active_same_type;
606  //find active with the same level
607  auto& sum_same_level = *std::find_if(sum_by_level.begin(), sum_by_level.end(), [&damping](auto& sum) {
608  return get<DampingLevel>(sum) == get<DampingLevel>(damping);
609  });
610  //remove active with the same type and level and add new one
611  get<MatrixIdx>(sum_same_level) += get<MatrixIdx>(damping) - get<MatrixIdx>(active_same_type).get();
612  //avoid negative values due to rounding error if e.g. a previous damping is lifted
613  get<MatrixIdx>(sum_same_level) = get<MatrixIdx>(sum_same_level).cwiseMax(0.);
614  get<MatrixIdx>(active_same_type) = get<MatrixIdx>(damping);
615  }
616  else {
617  //add new type to active
618  active_by_type.emplace_back(get<MatrixIdx>(damping), get<DampingLevel>(damping), get<DampingType>(damping));
619  //find damping with same level if existent
620  auto iter_sum_same_level = std::find_if(sum_by_level.begin(), sum_by_level.end(), [&damping](auto& sum) {
621  return get<DampingLevel>(sum) == get<DampingLevel>(damping);
622  });
623  if (iter_sum_same_level != sum_by_level.end()) {
624  //add to existing level
625  get<MatrixIdx>(*iter_sum_same_level) += get<MatrixIdx>(damping);
626  }
627  else {
628  //add new level
629  sum_by_level.emplace_back(get<MatrixIdx>(damping), get<DampingLevel>(damping));
630  }
631  }
632 }
