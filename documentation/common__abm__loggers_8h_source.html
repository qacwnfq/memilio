163 struct LogInfectionState : mio::LogAlways {
164  using Type = std::pair<mio::abm::TimePoint, Eigen::VectorXd>;
170  static Type log(const mio::abm::Simulation& sim)
171  {
172 
173  Eigen::VectorXd sum = Eigen::VectorXd::Zero(Eigen::Index(mio::abm::InfectionState::Count));
174  auto curr_time = sim.get_time();
175  PRAGMA_OMP(for)
176  for (auto&& location : sim.get_world().get_locations()) {
177  for (uint32_t inf_state = 0; inf_state < (int)mio::abm::InfectionState::Count; inf_state++) {
178  sum[inf_state] += location.get_subpopulation(curr_time, mio::abm::InfectionState(inf_state));
179  }
180  }
181  return std::make_pair(curr_time, sum);

209 template <class... Loggers>
210 struct DataWriterToMemoryDelta {
211  using Data = std::tuple<std::vector<typename Loggers::Type>...>;
212  template <class Logger>
219  static void add_record(const typename Logger::Type& t, Data& data)
220  {
221 
222  if (std::get<index_of_type_v<Logger, Loggers...>>(data).size() > 0) {
223  typename Logger::Type diff_vector{};
224  auto& current_state_vec = std::get<index_of_type_v<Logger, Loggers...>>(data).front();
225  for (auto i = 0; i < (int)current_state_vec.size(); i++) {
226  if (std::get<1>(t[i]) != std::get<1>(current_state_vec[i])) {
227  std::get<1>(current_state_vec[i]) = std::get<1>(t[i]);
228  diff_vector.push_back(t[i]);
229  }
230  }
231  std::get<index_of_type_v<Logger, Loggers...>>(data).push_back(diff_vector);
232  }
233  else {
234  std::get<index_of_type_v<Logger, Loggers...>>(data).push_back(
235  t); // We use the first entry as a reference for the current position.
236  std::get<index_of_type_v<Logger, Loggers...>>(data).push_back(t);
237  }
238  }
239 };
