191 template <class Derived>
192 auto RandomNumberGeneratorBase<Derived>::operator()() -> result_type
193 {
194  //generate a random sample using the Random123 library.
195  //Use another threefryNxR algorithm if larger or more random samples are needed than 64 bit.
196  auto self = static_cast<Derived*>(this);
197  auto c = static_cast<uint64_t>(self->get_counter().get());
198  auto k = static_cast<uint64_t>(self->get_key().get());
199  auto r = details::to_uint64(threefry2x32(details::to_r123_array(k), details::to_r123_array(c)));
200  self->increment_counter();
201  return r;
202 }

233 template <class UIntC, class UIntN, class CounterS>
234 Counter<UIntC> rng_totalsequence_counter(UIntN subsequence_idx, CounterS counter)
235 {
236  //use UIntC for variables because it's the biggest integer type in this function
237  static const UIntC BITS_PER_BYTE = 8;
238  static const UIntC C_BITS = sizeof(UIntC) * BITS_PER_BYTE;
239  static const UIntC S_BITS = sizeof(CounterS) * BITS_PER_BYTE;
240  static const UIntC N_BITS = C_BITS - S_BITS;
241 
242  static_assert(S_BITS < C_BITS, "Subsequence counter must be smaller than total sequence counter.");
243  static_assert(N_BITS <= C_BITS, "Subsequence index must not be bigger than total sequence counter.");
244  static_assert(N_BITS <= sizeof(UIntN) * BITS_PER_BYTE, "Subsequence index must be at least N bits");
245 
246  assert(UIntC(subsequence_idx) <= (UIntC(1) << N_BITS) && "Subsequence index is too large."); //(1 << N) is the same as (2^N)
247 
248  //N high bits: subsequence idx
249  //S low bits: subsequence counter
250  //=> C = N + S bits: total sequence counter
251  //example:
252  //subsequence index uint32_t(181) = 0x000000B5
253  //subsequence counter uint32_t(41309) = 0x0000A15D
254  //total sequence counter = 0x000000B50000A15D
255  const auto i = static_cast<UIntC>(subsequence_idx);
256  const auto s = static_cast<UIntC>(counter.get());
257  const auto c = (i << S_BITS) + s; //shift subsequence index to the high bits, add subsequence counter into low bits
258  return Counter<UIntC>{c};
259 }

616  template <class RNG>
617  result_type operator()(RNG& rng, param_type p)
618  {
619  auto weights = p.weights();
620  if (weights.size() <= 1) {
621  return 0;
622  }
623  auto sum = std::accumulate(weights.begin(), weights.end(), 0.0);
624  auto u =
625  std::uniform_real_distribution<double>()(rng, std::uniform_real_distribution<double>::param_type{0.0, sum});
626  auto intermediate_sum = 0.0;
627  for (size_t i = 0; i < weights.size(); ++i) {
628  intermediate_sum += weights.get_ptr()[i];
629  if (u < intermediate_sum) {
630  return i;
631  }
632  }
633  assert(false && "this should never happen.");
634  return result_type(-1);
635  }
636

638  param_type m_params;
