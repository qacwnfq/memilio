465 template <class IOContext, class F, class... T>
466 details::ApplyResultT<F, T...> apply(IOContext& io, F f, const IOResult<T>&... rs)
467 {
468  //store the status of each argument in an array to check them.
469  //it would be possible to write a recursive template that checks one argument
470  //after the other without copying each status. This would probably be slightly faster
471  //and easier to optimize for the compiler. but gdb crashes trying to resolve the
472  //very long symbol in case of many arguments. Successful results are very cheap to copy
473  //and slightly worse performance in the case of an error is probably acceptable.
474 
475  //check for errors in the arguments
476  IOStatus status[] = {(rs ? IOStatus{} : rs.error())...};
477  auto iter_err = std::find_if(std::begin(status), std::end(status), [](auto& s) {
478  return s.is_error();
479  });
480 
481  //evaluate f if all succesful
482  auto result =
483  iter_err == std::end(status) ? details::eval(f, rs...) : details::ApplyResultT<F, T...>(failure(*iter_err));
484 
485  if (!result) {
486  //save error in context
487  //if the error was generated by the applied function, the context hasn't seen it yet.
488  //this allows the context to fail fast and not continue to parse more.
489  io.set_error(result.error());
490  }
491  return result;
492 }

518 //recursive tuple deserialization for each tuple element
519 //read one tuple element after the other from the IOObject
520 //argument pack rs contains the elements that have been read already
521 template <class IOObj, class Tup, class... Ts>
522 std::enable_if_t<(sizeof...(Ts) == std::tuple_size<Tup>::value), IOResult<Tup>>
523 deserialize_tuple_element(IOObj& o, Tag<Tup>, const IOResult<Ts>&... rs)
524 {
525  //end of recursion
526  //number of arguments in rs is the same as the size of the tuple
527  //no more elements to read, so finalize the object
528  return mio::apply(
529  o,
530  [](const Ts&... ts) {
531  return Tup(ts...);
532  },
533  rs...);
534 }
535 template <class IOObj, class Tup, class... Ts>
536 std::enable_if_t<(sizeof...(Ts) < std::tuple_size<Tup>::value), IOResult<Tup>>
537 deserialize_tuple_element(IOObj& obj, Tag<Tup> tag, const IOResult<Ts>&... rs)
538 {
539  //get the next element of the tuple from the IO object
540  const size_t Idx = sizeof...(Ts);
541  auto r = obj.expect_element(make_tuple_element_name<Idx>(), Tag<std::tuple_element_t<Idx, Tup>>{});
542  //recurse, append the new element to the pack of arguments
543  return deserialize_tuple_element(obj, tag, rs..., r);
544 }

609 IOResult<M> deserialize_internal(IOContext& io, Tag<M> /*tag*/)
610 {
611  auto obj = io.expect_object("Matrix");
612  auto rows = obj.expect_element("Rows", Tag<Eigen::Index>{});
613  auto cols = obj.expect_element("Columns", Tag<Eigen::Index>{});
