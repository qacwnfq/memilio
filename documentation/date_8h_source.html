216 inline Date offset_date_by_days(Date date, int offset_days)
217 {
218  auto year = date.year;
219  auto month = date.month;
220  auto day = date.day;
221  assert(month > 0 && month < 13 && day > 0 && day <= get_month_length(date));
222 
223  if (day + offset_days > 0 && day + offset_days <= get_month_length(date)) {
224  return {year, month, day + offset_days};
225  }
226  else {
227  auto part_sum = calculate_partial_sum_of_months(date);
228 
229  int day_in_year = day + offset_days;
230  if (month > 1) {
231  // take month-2 since end of last month has to be found and due to start at 0 of C++ (against January=1 in date format)
232  day_in_year += part_sum[month - 2];
233  }
234 
235  if (day_in_year > 0 && day_in_year <= part_sum[11]) {
236  auto iter = std::find_if(part_sum.begin(), part_sum.end(), [day_in_year](auto s) {
237  return day_in_year <= s;
238  });
239  int i = static_cast<int>(iter - part_sum.begin());
240  return {year, i + 1, day_in_year - (i > 0 ? part_sum[static_cast<size_t>(i - 1)] : 0)};
241  }
242  else {
243  if (day_in_year > 0) {
244  return offset_date_by_days({year + 1, 1, 1}, day_in_year - part_sum[11] - 1);
245  }
246  else {
247  return offset_date_by_days({year - 1, 12, 31}, day_in_year);
248  }
249  }
250  }
251 }

284 inline int get_offset_in_days(Date date1, Date date2)
285 {
286  auto year1 = date1.year;
287  auto month1 = date1.month;
288  auto day1 = date1.day;
289 
290  auto year2 = date2.year;
291  auto month2 = date2.month;
292  auto day2 = date2.day;
293 
294  if (year1 == year2 && month1 == month2) {
295  return day1 - day2;
296  }
297  else {
298  int day_in_year1 = get_day_in_year(date1);
299  int day_in_year2 = get_day_in_year(date2);
300 
301  if (year1 < year2) {
302  int sum_days = 0;
303  for (int i = year1; i < year2; i++) {
304  sum_days += 365;
305  if (is_leap_year(i)) {
306  sum_days += 1;
307  }
308  }
309  return -(sum_days - day_in_year1) - day_in_year2;
310  }
311  else if (year1 > year2) {
312  int sum_days = 0;
313  for (int i = year2; i < year1; i++) {
314  sum_days += 365;
315  if (is_leap_year(i)) {
316  sum_days += 1;
317  }
318  }
319  return day_in_year1 + sum_days - day_in_year2;
320  }
321  else {
322 
323  return day_in_year1 - day_in_year2;
324  }
325  }
326 }
