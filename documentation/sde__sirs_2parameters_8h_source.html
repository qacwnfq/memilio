173  "and reset parameters.",
174  this->get<TimeImmune>(), 0.0);
175  return true;
176  }
177  if (this->get<TransmissionProbabilityOnContact>() < 0.0 ||
178  this->get<TransmissionProbabilityOnContact>() > 1.0) {
179  log_error(

181  this->get<TransmissionProbabilityOnContact>(), 0.0, 1.0);
