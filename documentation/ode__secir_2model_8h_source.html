73  void get_flows(Eigen::Ref<const Eigen::VectorXd> pop, Eigen::Ref<const Eigen::VectorXd> y, double t,
74  Eigen::Ref<Eigen::VectorXd> flows) const override
75  {
76  auto const& params = this->parameters;
77  AgeGroup n_agegroups = params.get_num_groups();
78 
79  ContactMatrixGroup const& contact_matrix = params.get<ContactPatterns>();
80 
81  auto icu_occupancy = 0.0;
82  auto test_and_trace_required = 0.0;
83  for (auto i = AgeGroup(0); i < n_agegroups; ++i) {
84  test_and_trace_required += (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i]) /
85  params.get<TimeInfectedNoSymptoms>()[i] *
86  this->populations.get_from(pop, {i, InfectionState::InfectedNoSymptoms});
87  icu_occupancy += this->populations.get_from(pop, {i, InfectionState::InfectedCritical});
88  }
89 
90  for (auto i = AgeGroup(0); i < n_agegroups; i++) {
91 
92  size_t Si = this->populations.get_flat_index({i, InfectionState::Susceptible});
93  size_t Ei = this->populations.get_flat_index({i, InfectionState::Exposed});
94  size_t INSi = this->populations.get_flat_index({i, InfectionState::InfectedNoSymptoms});
95  size_t INSCi = this->populations.get_flat_index({i, InfectionState::InfectedNoSymptomsConfirmed});
96  size_t ISyi = this->populations.get_flat_index({i, InfectionState::InfectedSymptoms});
97  size_t ISyCi = this->populations.get_flat_index({i, InfectionState::InfectedSymptomsConfirmed});
98  size_t ISevi = this->populations.get_flat_index({i, InfectionState::InfectedSevere});
99  size_t ICri = this->populations.get_flat_index({i, InfectionState::InfectedCritical});
100 
101  for (auto j = AgeGroup(0); j < n_agegroups; j++) {
102  size_t Sj = this->populations.get_flat_index({j, InfectionState::Susceptible});
103  size_t Ej = this->populations.get_flat_index({j, InfectionState::Exposed});
104  size_t INSj = this->populations.get_flat_index({j, InfectionState::InfectedNoSymptoms});
105  size_t ISyj = this->populations.get_flat_index({j, InfectionState::InfectedSymptoms});
106  size_t ISevj = this->populations.get_flat_index({j, InfectionState::InfectedSevere});
107  size_t ICrj = this->populations.get_flat_index({j, InfectionState::InfectedCritical});
108  size_t Rj = this->populations.get_flat_index({j, InfectionState::Recovered});
109 
110  //symptomatic are less well quarantined when testing and tracing is overwhelmed so they infect more people
111  auto riskFromInfectedSymptomatic = smoother_cosine(
112  test_and_trace_required, params.get<TestAndTraceCapacity>(), params.get<TestAndTraceCapacity>() * 5,
113  params.get<RiskOfInfectionFromSymptomatic>()[j],
114  params.get<MaxRiskOfInfectionFromSymptomatic>()[j]);
115 
116  // effective contact rate by contact rate between groups i and j and damping j
117  double season_val =
118  (1 + params.get<Seasonality>() *
119  sin(3.141592653589793 * (std::fmod((params.get<StartDay>() + t), 365.0) / 182.5 + 0.5)));
120  double cont_freq_eff =
121  season_val * contact_matrix.get_matrix_at(t)(static_cast<Eigen::Index>((size_t)i),
122  static_cast<Eigen::Index>((size_t)j));
123  double Nj =
124  pop[Sj] + pop[Ej] + pop[INSj] + pop[ISyj] + pop[ISevj] + pop[ICrj] + pop[Rj]; // without died people
125  double divNj = 1.0 / Nj; // precompute 1.0/Nj
126  double dummy_S = y[Si] * cont_freq_eff * divNj * params.get<TransmissionProbabilityOnContact>()[i] *
127  (params.get<RelativeTransmissionNoSymptoms>()[j] * pop[INSj] +
128  riskFromInfectedSymptomatic * pop[ISyj]);
129 
130  // Susceptible -> Exposed
131  flows[get_flat_flow_index<InfectionState::Susceptible, InfectionState::Exposed>({i})] += dummy_S;
132  }
133 
134  // ICU capacity shortage is close
135  double criticalPerSevereAdjusted =
136  smoother_cosine(icu_occupancy, 0.90 * params.get<ICUCapacity>(), params.get<ICUCapacity>(),
137  params.get<CriticalPerSevere>()[i], 0);
138 
139  double deathsPerSevereAdjusted = params.get<CriticalPerSevere>()[i] - criticalPerSevereAdjusted;
140 
141  // Exposed -> InfectedNoSymptoms
142  flows[get_flat_flow_index<InfectionState::Exposed, InfectionState::InfectedNoSymptoms>({i})] =
143  (1 / params.get<TimeExposed>()[i]) * y[Ei];
144 
145  // InfectedNoSymptoms -> InfectedSymptoms / Recovered
146  flows[get_flat_flow_index<InfectionState::InfectedNoSymptoms, InfectionState::InfectedSymptoms>({i})] =
147  (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i]) * (1 / params.get<TimeInfectedNoSymptoms>()[i]) *
148  y[INSi];
149  flows[get_flat_flow_index<InfectionState::InfectedNoSymptoms, InfectionState::Recovered>({i})] =
150  params.get<RecoveredPerInfectedNoSymptoms>()[i] * (1 / params.get<TimeInfectedNoSymptoms>()[i]) *
151  y[INSi];
152 
153  // InfectedNoSymptomsConfirmed -> InfectedSymptomsConfirmed / Recovered
154  flows[get_flat_flow_index<InfectionState::InfectedNoSymptomsConfirmed,
155  InfectionState::InfectedSymptomsConfirmed>({i})] =
156  (1 - params.get<RecoveredPerInfectedNoSymptoms>()[i]) * (1 / params.get<TimeInfectedNoSymptoms>()[i]) *
157  y[INSCi];
158  flows[get_flat_flow_index<InfectionState::InfectedNoSymptomsConfirmed, InfectionState::Recovered>({i})] =
159  params.get<RecoveredPerInfectedNoSymptoms>()[i] * (1 / params.get<TimeInfectedNoSymptoms>()[i]) *
160  y[INSCi];
161 
162  // InfectedSymptoms -> InfectedSevere / Recovered
163  flows[get_flat_flow_index<InfectionState::InfectedSymptoms, InfectionState::InfectedSevere>({i})] =
164  params.get<SeverePerInfectedSymptoms>()[i] / params.get<TimeInfectedSymptoms>()[i] * y[ISyi];
165  flows[get_flat_flow_index<InfectionState::InfectedSymptoms, InfectionState::Recovered>({i})] =
166  (1 - params.get<SeverePerInfectedSymptoms>()[i]) / params.get<TimeInfectedSymptoms>()[i] * y[ISyi];
167 
168  // InfectedSymptomsConfirmed -> InfectedSevere / Recovered
169  flows[get_flat_flow_index<InfectionState::InfectedSymptomsConfirmed, InfectionState::InfectedSevere>({i})] =
170  params.get<SeverePerInfectedSymptoms>()[i] / params.get<TimeInfectedSymptoms>()[i] * y[ISyCi];
171  flows[get_flat_flow_index<InfectionState::InfectedSymptomsConfirmed, InfectionState::Recovered>({i})] =
172  (1 - params.get<SeverePerInfectedSymptoms>()[i]) / params.get<TimeInfectedSymptoms>()[i] * y[ISyCi];
173 
174  // InfectedSevere -> InfectedCritical / Recovered / Dead
175  flows[get_flat_flow_index<InfectionState::InfectedSevere, InfectionState::InfectedCritical>({i})] =
176  criticalPerSevereAdjusted / params.get<TimeInfectedSevere>()[i] * y[ISevi];
177  flows[get_flat_flow_index<InfectionState::InfectedSevere, InfectionState::Recovered>({i})] =
178  (1 - params.get<CriticalPerSevere>()[i]) / params.get<TimeInfectedSevere>()[i] * y[ISevi];
179  flows[get_flat_flow_index<InfectionState::InfectedSevere, InfectionState::Dead>({i})] =
180  deathsPerSevereAdjusted / params.get<TimeInfectedSevere>()[i] * y[ISevi];
181 
182  // InfectedCritical -> Dead / Recovered
183  flows[get_flat_flow_index<InfectionState::InfectedCritical, InfectionState::Dead>({i})] =
184  params.get<DeathsPerCritical>()[i] / params.get<TimeInfectedCritical>()[i] * y[ICri];
185  flows[get_flat_flow_index<InfectionState::InfectedCritical, InfectionState::Recovered>({i})] =
186  (1 - params.get<DeathsPerCritical>()[i]) / params.get<TimeInfectedCritical>()[i] * y[ICri];
187  }
188  }

249  Simulation(Model const& model, double t0 = 0., double dt = 0.1)
250  : BaseT(model, t0, dt)
251  , m_t_last_npi_check(t0)
252  {
253  }

262  Eigen::Ref<Eigen::VectorXd> advance(double tmax)
263  {
264  auto& dyn_npis = this->get_model().parameters.template get<DynamicNPIsInfectedSymptoms>();
265  auto& contact_patterns = this->get_model().parameters.template get<ContactPatterns>();
266  if (dyn_npis.get_thresholds().size() > 0) {
267  auto t = BaseT::get_result().get_last_time();
268  const auto dt = dyn_npis.get_interval().get();
269 
270  while (t < tmax) {
271  auto dt_eff = std::min({dt, tmax - t, m_t_last_npi_check + dt - t});
272 
273  BaseT::advance(t + dt_eff);
274  t = t + dt_eff;
275 
276  if (floating_point_greater_equal(t, m_t_last_npi_check + dt)) {
277  auto inf_rel = get_infections_relative(*this, t, this->get_result().get_last_value()) *
278  dyn_npis.get_base_value();
279  auto exceeded_threshold = dyn_npis.get_max_exceeded_threshold(inf_rel);
280  if (exceeded_threshold != dyn_npis.get_thresholds().end() &&
281  (exceeded_threshold->first > m_dynamic_npi.first ||
282  t > double(m_dynamic_npi.second))) { //old npi was weaker or is expired
283  auto t_end = mio::SimulationTime(t + double(dyn_npis.get_duration()));
284  m_dynamic_npi = std::make_pair(exceeded_threshold->first, t_end);
285  mio::implement_dynamic_npis(contact_patterns.get_cont_freq_mat(), exceeded_threshold->second,
286  SimulationTime(t), t_end, [](auto& g) {
287  return mio::make_contact_damping_matrix(g);
288  });
289  }
290 
291  m_t_last_npi_check = t;
292  }
293  }
294 
295  return this->get_result().get_last_value();
296  }
297  else {
298  return BaseT::advance(tmax);
299  }
300  }
301 
302 private:
303  double m_t_last_npi_check;
304  std::pair<double, SimulationTime> m_dynamic_npi = {-std::numeric_limits<double>::max(), mio::SimulationTime(0)};
305 };

343 double get_infections_relative(const Simulation<Base>& sim, double /*t*/, const Eigen::Ref<const Eigen::VectorXd>& y)
344 {
345  double sum_inf = 0;
346  for (auto i = AgeGroup(0); i < sim.get_model().parameters.get_num_groups(); ++i) {
347  sum_inf += sim.get_model().populations.get_from(y, {i, InfectionState::InfectedSymptoms});
348  }
349  auto inf_rel = sum_inf / sim.get_model().populations.get_total();
350 
351  return inf_rel;

362 template <class Base>
363 IOResult<ScalarType> get_reproduction_number(size_t t_idx, const Simulation<Base>& sim)
364 {
365 
366  if (!(t_idx < static_cast<size_t>(sim.get_result().get_num_time_points()))) {
367  return mio::failure(mio::StatusCode::OutOfRange, "t_idx is not a valid index for the TimeSeries");
368  }
369 
370  auto const& params = sim.get_model().parameters;
371  const size_t num_groups = (size_t)sim.get_model().parameters.get_num_groups();
372  //The infected compartments in the SECIR Model are: Exposed, Carrier, Infected, Hospitalized, ICU in respective agegroups
373  const size_t num_infected_compartments = 5;
374  const size_t total_infected_compartments = num_infected_compartments * num_groups;
375  const double pi = std::acos(-1);
376 
377  //F encodes new Infections and V encodes transition times in the next-generation matrix calculation of R_t
378  Eigen::MatrixXd F(total_infected_compartments, total_infected_compartments);
379  Eigen::MatrixXd V(total_infected_compartments, total_infected_compartments);
380  F = Eigen::MatrixXd::Zero(total_infected_compartments,
381  total_infected_compartments); //Initialize matrices F and V with zeroes
382  V = Eigen::MatrixXd::Zero(total_infected_compartments, total_infected_compartments);
383 
384  auto test_and_trace_required = 0.0;
385  auto icu_occupancy = 0.0;
386  for (auto i = AgeGroup(0); i < (mio::AgeGroup)num_groups; ++i) {
387  test_and_trace_required +=
388  (1 - params.template get<RecoveredPerInfectedNoSymptoms>()[i]) /
389  params.template get<TimeInfectedNoSymptoms>()[i] *
390  sim.get_result().get_value(
391  t_idx)[sim.get_model().populations.get_flat_index({i, InfectionState::InfectedNoSymptoms})];
392  icu_occupancy += sim.get_result().get_value(
393  t_idx)[sim.get_model().populations.get_flat_index({i, InfectionState::InfectedCritical})];
394  }
395 
396  double season_val = (1 + params.template get<Seasonality>() *
397  sin(pi * (std::fmod((sim.get_model().parameters.template get<StartDay>() +
398  sim.get_result().get_time(t_idx)),
399  365.0) /
400  182.5 +
401  0.5)));
402  ContactMatrixGroup const& contact_matrix = sim.get_model().parameters.template get<ContactPatterns>();
403 
404  Eigen::MatrixXd cont_freq_eff(num_groups, num_groups);
405  Eigen::MatrixXd riskFromInfectedSymptomatic_derivatives(num_groups, num_groups);
406  Eigen::VectorXd divN(num_groups);
407  Eigen::VectorXd riskFromInfectedSymptomatic(num_groups);
408 
409  for (mio::AgeGroup k = 0; k < (mio::AgeGroup)num_groups; k++) {
410  double temp = sim.get_result().get_value(
411  t_idx)[sim.get_model().populations.get_flat_index({k, InfectionState::Susceptible})] +
412  sim.get_result().get_value(
413  t_idx)[sim.get_model().populations.get_flat_index({k, InfectionState::Exposed})] +
414  sim.get_result().get_value(
415  t_idx)[sim.get_model().populations.get_flat_index({k, InfectionState::InfectedNoSymptoms})] +
416  sim.get_result().get_value(
417  t_idx)[sim.get_model().populations.get_flat_index({k, InfectionState::InfectedSymptoms})] +
418  sim.get_result().get_value(
419  t_idx)[sim.get_model().populations.get_flat_index({k, InfectionState::InfectedSevere})] +
420  sim.get_result().get_value(
421  t_idx)[sim.get_model().populations.get_flat_index({k, InfectionState::InfectedCritical})] +
422  sim.get_result().get_value(
423  t_idx)[sim.get_model().populations.get_flat_index({k, InfectionState::Recovered})];
424  if (temp == 0) {
425  temp = 1;
426  }
427  divN[(size_t)k] = 1 / temp;
428 
429  riskFromInfectedSymptomatic[(size_t)k] = smoother_cosine(
430  test_and_trace_required, params.template get<TestAndTraceCapacity>(),
431  (params.template get<TestAndTraceCapacity>()) * 5, params.template get<RiskOfInfectionFromSymptomatic>()[k],
432  params.template get<MaxRiskOfInfectionFromSymptomatic>()[k]);
433 
434  for (mio::AgeGroup l = 0; l < (mio::AgeGroup)num_groups; l++) {
435  if (test_and_trace_required < params.template get<TestAndTraceCapacity>() ||
436  test_and_trace_required > 5 * params.template get<TestAndTraceCapacity>()) {
437  riskFromInfectedSymptomatic_derivatives((size_t)k, (size_t)l) = 0;
438  }
439  else {
440  riskFromInfectedSymptomatic_derivatives((size_t)k, (size_t)l) =
441  -0.5 * pi *
442  (params.template get<MaxRiskOfInfectionFromSymptomatic>()[k] -
443  params.template get<RiskOfInfectionFromSymptomatic>()[k]) /
444  (4 * params.template get<TestAndTraceCapacity>()) *
445  (1 - params.template get<RecoveredPerInfectedNoSymptoms>()[l]) /
446  params.template get<TimeInfectedNoSymptoms>()[l] *
447  std::sin(pi / (4 * params.template get<TestAndTraceCapacity>()) *
448  (test_and_trace_required - params.template get<TestAndTraceCapacity>()));
449  }
450  }
451 
452  for (Eigen::Index l = 0; l < (Eigen::Index)num_groups; l++) {
453  cont_freq_eff(l, (size_t)k) =
454  season_val * contact_matrix.get_matrix_at(static_cast<double>(t_idx))(
455  static_cast<Eigen::Index>((size_t)l), static_cast<Eigen::Index>((size_t)k));
456  }
457  }
458 
459  //Check criterion if matrix V will be invertible by checking if subblock J is invertible
460  Eigen::MatrixXd J(num_groups, num_groups);
461  J = Eigen::MatrixXd::Zero(num_groups, num_groups);
462  for (size_t i = 0; i < num_groups; i++) {
463  J(i, i) = 1 / (params.template get<TimeInfectedCritical>()[(mio::AgeGroup)i]);
464 
465  if (!(icu_occupancy < 0.9 * params.template get<ICUCapacity>() ||
466  icu_occupancy > (double)(params.template get<ICUCapacity>()))) {
467  for (size_t j = 0; j < num_groups; j++) {
468  J(i, j) -= sim.get_result().get_value(t_idx)[sim.get_model().populations.get_flat_index(
469  {(mio::AgeGroup)i, InfectionState::InfectedSevere})] /
470  params.template get<TimeInfectedSevere>()[(mio::AgeGroup)i] * 5 *
471  params.template get<CriticalPerSevere>()[(mio::AgeGroup)i] * pi /
472  (params.template get<ICUCapacity>()) *
473  std::sin(pi / (0.1 * params.template get<ICUCapacity>()) *
474  (icu_occupancy - 0.9 * params.template get<ICUCapacity>()));
475  }
476  }
477  }
478 
479  //Check, if J is invertible
480  if (J.determinant() == 0) {
481  return mio::failure(mio::StatusCode::UnknownError, "Matrix V is not invertible");
482  }
483 
484  //Initialize the matrix F
485  for (size_t i = 0; i < num_groups; i++) {
486  for (size_t j = 0; j < num_groups; j++) {
487 
488  double temp = 0;
489  for (Eigen::Index k = 0; k < (Eigen::Index)num_groups; k++) {
490  temp += cont_freq_eff(i, k) *
491  sim.get_result().get_value(t_idx)[sim.get_model().populations.get_flat_index(
492  {(mio::AgeGroup)k, InfectionState::InfectedSymptoms})] *
493  riskFromInfectedSymptomatic_derivatives(k, j) * divN[k];
494  }
495 
496  F(i, j + num_groups) =
497  sim.get_result().get_value(t_idx)[sim.get_model().populations.get_flat_index(
498  {(mio::AgeGroup)i, InfectionState::Susceptible})] *
499  params.template get<TransmissionProbabilityOnContact>()[(mio::AgeGroup)i] *
500  (cont_freq_eff(i, j) * params.template get<RelativeTransmissionNoSymptoms>()[(mio::AgeGroup)j] *
501  divN[(size_t)j] +
502  temp);
503  }
504 
505  for (size_t j = 0; j < num_groups; j++) {
506  F(i, j + 2 * num_groups) = sim.get_result().get_value(t_idx)[sim.get_model().populations.get_flat_index(
507  {(mio::AgeGroup)i, InfectionState::Susceptible})] *
508  params.template get<TransmissionProbabilityOnContact>()[(mio::AgeGroup)i] *
509  cont_freq_eff(i, j) * riskFromInfectedSymptomatic[(size_t)j] * divN[(size_t)j];
510  }
511  }
512 
513  //Initialize the matrix V
514  for (Eigen::Index i = 0; i < (Eigen::Index)num_groups; i++) {
515 
516  double criticalPerSevereAdjusted = smoother_cosine(
517  icu_occupancy, 0.90 * params.template get<ICUCapacity>(), params.template get<ICUCapacity>(),
518  params.template get<CriticalPerSevere>()[(mio::AgeGroup)i], 0);
519 
520  V(i, i) = 1 / params.template get<TimeExposed>()[(mio::AgeGroup)i];
521  V(i + num_groups, i) = -1 / params.template get<TimeExposed>()[(mio::AgeGroup)i];
522  V(i + num_groups, i + num_groups) = 1 / params.template get<TimeInfectedNoSymptoms>()[(mio::AgeGroup)i];
523  V(i + 2 * num_groups, i + num_groups) =
524  -(1 - params.template get<RecoveredPerInfectedNoSymptoms>()[(mio::AgeGroup)i]) /
525  params.template get<TimeInfectedNoSymptoms>()[(mio::AgeGroup)i];
526  V(i + 2 * num_groups, i + 2 * num_groups) = (1 / params.template get<TimeInfectedSymptoms>()[(mio::AgeGroup)i]);
527  V(i + 3 * num_groups, i + 2 * num_groups) =
528  -params.template get<SeverePerInfectedSymptoms>()[(mio::AgeGroup)i] /
529  params.template get<TimeInfectedSymptoms>()[(mio::AgeGroup)i];
530  V(i + 3 * num_groups, i + 3 * num_groups) = 1 / (params.template get<TimeInfectedSevere>()[(mio::AgeGroup)i]);
531  V(i + 4 * num_groups, i + 3 * num_groups) =
532  -criticalPerSevereAdjusted / (params.template get<TimeInfectedSevere>()[(mio::AgeGroup)i]);
533 
534  for (size_t j = 0; j < num_groups; j++) {
535  V(i + 4 * num_groups, j + 4 * num_groups) = J(i, j);
536  }
537  }
538 
539  V = V.inverse();
540 
541  //Compute F*V
542  Eigen::MatrixXd NextGenMatrix(num_infected_compartments * num_groups, 5 * num_groups);
543  NextGenMatrix = F * V;
544 
545  //Compute the largest eigenvalue in absolute value
546  Eigen::ComplexEigenSolver<Eigen::MatrixXd> ces;
547 
548  ces.compute(NextGenMatrix);
549  const Eigen::VectorXcd eigen_vals = ces.eigenvalues();
550 
551  Eigen::VectorXd eigen_vals_abs;
552  eigen_vals_abs.resize(eigen_vals.size());
553 
554  for (int i = 0; i < eigen_vals.size(); i++) {
555  eigen_vals_abs[i] = std::abs(eigen_vals[i]);
556  }
557  return mio::success(eigen_vals_abs.maxCoeff());
558 }

617 template <class Base = mio::Simulation<Model>>
618 auto get_migration_factors(const Simulation<Base>& sim, double /*t*/, const Eigen::Ref<const Eigen::VectorXd>& y)
619 {
620  auto& params = sim.get_model().parameters;
621  //parameters as arrays
622  auto&& p_asymp = params.template get<RecoveredPerInfectedNoSymptoms>().array().template cast<double>();
623  auto&& p_inf = params.template get<RiskOfInfectionFromSymptomatic>().array().template cast<double>();
624  auto&& p_inf_max = params.template get<MaxRiskOfInfectionFromSymptomatic>().array().template cast<double>();
625  //slice of InfectedNoSymptoms
626  auto y_INS = slice(y, {Eigen::Index(InfectionState::InfectedNoSymptoms),
627  Eigen::Index(size_t(params.get_num_groups())), Eigen::Index(InfectionState::Count)});
628 
629  //compute isolation, same as infection risk from main model
630  auto test_and_trace_required =
631  ((1 - p_asymp) / params.template get<TimeInfectedNoSymptoms>().array().template cast<double>() * y_INS.array())
632  .sum();
633  auto test_and_trace_capacity = double(params.template get<TestAndTraceCapacity>());
634  auto riskFromInfectedSymptomatic = smoother_cosine(test_and_trace_required, test_and_trace_capacity,
635  test_and_trace_capacity * 5, p_inf.matrix(), p_inf_max.matrix());
636 
637  //set factor for infected
638  auto factors = Eigen::VectorXd::Ones(y.rows()).eval();
639  slice(factors, {Eigen::Index(InfectionState::InfectedSymptoms), Eigen::Index(size_t(params.get_num_groups())),
640  Eigen::Index(InfectionState::Count)})
641  .array() = riskFromInfectedSymptomatic;
642  return factors;
643 }

645 template <class Base = mio::Simulation<Model>>
646 auto test_commuters(Simulation<Base>& sim, Eigen::Ref<Eigen::VectorXd> migrated, double time)
647 {
648  auto& model = sim.get_model();
649  auto nondetection = 1.0;
650  if (time >= model.parameters.get_start_commuter_detection() &&
651  time < model.parameters.get_end_commuter_detection()) {
652  nondetection = (double)model.parameters.get_commuter_nondetection();
653  }
654  for (auto i = AgeGroup(0); i < model.parameters.get_num_groups(); ++i) {
655  auto INSi = model.populations.get_flat_index({i, InfectionState::InfectedNoSymptoms});
656  auto INSCi = model.populations.get_flat_index({i, InfectionState::InfectedNoSymptomsConfirmed});
657  auto ISyi = model.populations.get_flat_index({i, InfectionState::InfectedSymptoms});
658  auto ISyCi = model.populations.get_flat_index({i, InfectionState::InfectedSymptomsConfirmed});
659 
660  //put detected commuters in their own compartment so they don't contribute to infections in their home node
661  sim.get_result().get_last_value()[INSi] -= migrated[INSi] * (1 - nondetection);
662  sim.get_result().get_last_value()[INSCi] += migrated[INSi] * (1 - nondetection);
663  sim.get_result().get_last_value()[ISyi] -= migrated[ISyi] * (1 - nondetection);
664  sim.get_result().get_last_value()[ISyCi] += migrated[ISyi] * (1 - nondetection);
665 
666  //reduce the number of commuters
667  migrated[ISyi] *= nondetection;
668  migrated[INSi] *= nondetection;
669  }
670 }
