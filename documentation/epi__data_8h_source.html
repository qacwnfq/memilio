310 inline void get_rki_age_interpolation_coefficients(const std::vector<double>& age_ranges,
311  std::vector<std::vector<double>>& interpolation,
312  std::vector<bool>& carry_over)
313 {
314  std::array<double, 6> param_ranges = {5., 10., 20., 25., 20., 20.};
315  assert(param_ranges.size() == ConfirmedCasesDataEntry::age_group_names.size() &&
316  "Number of RKI age groups does not match number of age ranges.");
317 
318  //counter for parameter age groups
319  size_t counter = 0;
320 
321  //residual of param age groups
322  double res = 0.0;
323  for (size_t i = 0; i < age_ranges.size(); i++) {
324 
325  // if current param age group didn't fit into previous rki age group, transfer residual to current age group
326  if (res < 0) {
327  interpolation[i].push_back(std::min(-res / age_ranges[i], 1.0));
328  }
329 
330  if (counter < param_ranges.size() - 1) {
331  res += age_ranges[i];
332  if (std::abs(res) < age_ranges[i]) {
333  counter++;
334  }
335  // iterate over param age groups while there is still room in the current rki age group
336  while (res > 0) {
337  res -= param_ranges[counter];
338  interpolation[i].push_back((param_ranges[counter] + std::min(res, 0.0)) / age_ranges[i]);
339  if (res >= 0) {
340  counter++;
341  }
342  }
343  if (res < 0) {
344  carry_over.push_back(true);
345  }
346  else if (res == 0) {
347  carry_over.push_back(false);
348  }
349  }
350  // if last param age group is reached
351  else {
352  interpolation[i].push_back((age_ranges[i] + res) / age_ranges[i]);
353  if (res < 0 || counter == 0) {
354  carry_over.push_back(true);
355  }
356  else if (res == 0) {
357  carry_over.push_back(false);
358  }
359  res = 0;
360  }
361  }
362 }

364 inline std::vector<PopulationDataEntry>
365 interpolate_to_rki_age_groups(const std::vector<PopulationDataEntry>& population_data)
366 {
367  std::vector<double> age_ranges = {3., 3., 9., 3., 7., 5., 10., 10., 15., 10., 25.};
368  std::vector<std::vector<double>> coefficients{age_ranges.size()};
369  std::vector<bool> carry_over{};
370  get_rki_age_interpolation_coefficients(age_ranges, coefficients, carry_over);
371 
372  std::vector<PopulationDataEntry> interpolated{population_data};
373  for (auto region_entry_idx = size_t(0); region_entry_idx < population_data.size(); ++region_entry_idx) {
374  interpolated[region_entry_idx].population =
375  CustomIndexArray<double, AgeGroup>(AgeGroup(ConfirmedCasesDataEntry::age_group_names.size()), 0.0);
376 
377  size_t interpolated_age_idx = 0;
378  for (size_t age_idx = 0; age_idx < coefficients.size(); age_idx++) {
379  for (size_t coeff_idx = 0; coeff_idx < coefficients[age_idx].size(); coeff_idx++) {
380  interpolated[region_entry_idx].population[AgeGroup(interpolated_age_idx)] +=
381  coefficients[age_idx][coeff_idx] * population_data[region_entry_idx].population[AgeGroup(age_idx)];
382  if (coeff_idx < coefficients[age_idx].size() - 1 || !carry_over[age_idx]) {
383  interpolated_age_idx++;
384  }
385  }
386  }
387  }
388 
389  return interpolated;
390 }
