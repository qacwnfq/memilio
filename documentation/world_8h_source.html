74  World(const World& other)
75  : parameters(other.parameters)
76  , m_persons()
77  , m_locations()
78  , m_trip_list(other.m_trip_list)
79  , m_cemetery_id(add_location(LocationType::Cemetery))
80  {
81  for (auto& origin_loc : other.get_locations()) {
82  if (origin_loc.get_type() != LocationType::Cemetery) {
83  // Copy a location
84  m_locations.emplace_back(
85  std::make_unique<Location>(origin_loc.copy_location_without_persons(parameters.get_num_groups())));
86  }
87  for (auto& person : other.get_persons()) {
88  // If a person is in this location, copy this person and add it to this location.
89  if (person.get_location() == origin_loc) {
90  LocationId origin_id = {origin_loc.get_index(), origin_loc.get_type()};
91  m_persons.push_back(
92  std::make_unique<Person>(person.copy_person(get_individualized_location(origin_id))));
93  }
94  }
95  }
96  use_migration_rules(other.m_use_migration_rules);
97  }

108  template <class IOContext>
109  void serialize(IOContext& io) const
110  {
111  auto obj = io.create_object("World");
112  obj.add_element("num_agegroups", parameters.get_num_groups());
113  std::vector<Trip> trips;
114  TripList trip_list = get_trip_list();
115  for (size_t i = 0; i < trip_list.num_trips(false); i++) {
116  trips.push_back(trip_list.get_next_trip(false));
117  trip_list.increase_index();
118  }
119  trip_list.reset_index();
120  for (size_t i = 0; i < trip_list.num_trips(true); i++) {
121  trips.push_back(trip_list.get_next_trip(true));
122  trip_list.increase_index();
123  }
124  obj.add_list("trips", trips.begin(), trips.end());
125  obj.add_list("locations", get_locations().begin(), get_locations().end());
126  obj.add_list("persons", get_persons().begin(), get_persons().end());
127  obj.add_element("use_migration_rules", m_use_migration_rules);
128  }
