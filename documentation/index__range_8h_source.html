119  template <size_t I = MultiIndex::size - 1>
120  inline void increment_index()
121  {
122  assert(mio::get<I>(m_dims).get() > 0 && "All dimensions must be positive.");
123  assert(mio::get<I>(m_index) < mio::get<I>(m_dims) && "Index out of bounds.");
124 
125  if constexpr (I > 0) {
126  // increment first, then do a carry-over if necessary
127  ++mio::get<I>(m_index);
128  if (mio::get<I>(m_index) == mio::get<I>(m_dims)) {
129  mio::get<I>(m_index) = mio::get<I>(m_index).Zero();
130  increment_index<I - 1>();
131  }
132  }
133  else {
134  ++mio::get<0>(m_index);
135  // no carry check for the most significant index
136  }
137  }
