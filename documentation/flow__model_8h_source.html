72 template <class Comp, class Pop, class Params, class Flows>
73 class FlowModel : public CompartmentalModel<Comp, Pop, Params>
74 {
75  using PopIndex = typename Pop::Index;
76  // FlowIndex is the same as PopIndex without the category Comp. It is used as argument type for
77  // get_flat_flow_index, since a flow is used to determine only the compartment (i.e. Index<Comp>) for the
78  // population of the model.
79  // The remaining indices (those contained in FlowIndex) must still be provided to get an entry of population.
80  // This approach only works with exactly one category of type Comp in PopIndex (hence the assertion below).
81  using FlowIndex = details::filtered_index_t<Comp, Index, PopIndex>;
82  // Enforce that Comp is a unique Category of PopIndex, since we use Flows (via their source/target) to provide
83  // Comp indices for the population.
84  static_assert(FlowIndex::size == PopIndex::size - 1, "Compartments must be used exactly once as population index.");

111  void get_derivatives(Eigen::Ref<const Eigen::VectorXd> flows, Eigen::Ref<Eigen::VectorXd> dydt) const
112  {
113  // set dydt to 0, then iteratively add all flow contributions
114  dydt.setZero();
115  if constexpr (std::is_same_v<FlowIndex, Index<>>) {
116  // special case where PopIndex only contains Comp, hence FlowIndex has no dimensions to iterate over
117  get_rhs_impl(flows, dydt, Index<>{});
118  }
119  else {
120  for (FlowIndex I : make_index_range(reduce_index<FlowIndex>(this->populations.size()))) {
121  get_rhs_impl(flows, dydt, I);
122  }
123  }
124  }

180  template <Comp Source, Comp Target>
181  size_t get_flat_flow_index(const FlowIndex& indices) const
182  {
183  if constexpr (std::is_same_v<FlowIndex, Index<>>) {
184  return get_flat_flow_index<Source, Target>();
185  }
186  else {

218  template <size_t I = 0>
219  inline void get_rhs_impl(Eigen::Ref<const Eigen::VectorXd> flows, Eigen::Ref<Eigen::VectorXd> rhs,
220  const FlowIndex& index) const
221  {
222  using Flow = type_at_index_t<I, Flows>;
223  const auto flat_flow_index = get_flat_flow_index<Flow::source, Flow::target>(index);
224  const auto flat_source_population =
225  this->populations.get_flat_index(extend_index<PopIndex>(index, (size_t)Flow::source));
226  const auto flat_target_population =
227  this->populations.get_flat_index(extend_index<PopIndex>(index, (size_t)Flow::target));
228  rhs[flat_source_population] -= flows[flat_flow_index]; // subtract outflow from source compartment
229  rhs[flat_target_population] += flows[flat_flow_index]; // add outflow to target compartment
230  // handle next flow (if there is one)
231  if constexpr (I + 1 < Flows::size()) {
232  get_rhs_impl<I + 1>(flows, rhs, index);
233  }
234  }
