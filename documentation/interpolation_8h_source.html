55 {
56  // If the vector is empty or has only 1 node, throw an error
57  if (vector.empty() || vector.size() == 1) {

59  return 0.0;

62  // Find the corresponding upper position of the node in the data set

64  [](double value, const std::pair<X, Y>& node) {

66  }) -
67  vector.begin();
68 
69  if (upper_pos == 0) {

72
