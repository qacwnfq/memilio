56  Eigen::Ref<Eigen::VectorXd> advance(double tmax)
57  {
58  // the derivfunktion (i.e. the lambda passed to m_integrator.advance below) requires that there are at least
59  // as many entries in m_flow_result as in Base::m_result
60  assert(m_flow_result.get_num_time_points() == this->get_result().get_num_time_points());
61  auto result = this->get_ode_integrator().advance(
62  [this](auto&& flows, auto&& t, auto&& dflows_dt) {
63  const auto& pop_result = this->get_result();
64  const auto& model = this->get_model();
65  // compute current population
66  // flows contains the accumulated outflows of each compartment for each target compartment at time t.
67  // Using that the ODEs are linear expressions of the flows, get_derivatives can compute the total change
68  // in population from t0 to t.
69  // To incorporate external changes to the last values of pop_result (e.g. by applying mobility), we only
70  // calculate the change in population starting from the last available time point in m_result, instead
71  // of starting at t0. To do that, the following difference of flows is used.
72  model.get_derivatives(flows - m_flow_result.get_value(pop_result.get_num_time_points() - 1),
73  m_pop); // note: overwrites values in pop
74  // add the "initial" value of the ODEs (using last available time point in pop_result)
75  // If no changes were made to the last value in m_result outside of FlowSimulation, the following
76  // line computes the same as `model.get_derivatives(flows, x); x += model.get_initial_values();`.
77  m_pop += pop_result.get_last_value();
78  // compute the current change in flows with respect to the current population
79  dflows_dt.setZero();
80  model.get_flows(m_pop, m_pop, t, dflows_dt); // this result is used by the integrator
81  },
82  tmax, this->get_dt(), m_flow_result);
83  compute_population_results();
84  return result;
85  }

115  void compute_population_results()
116  {
117  const auto& flows = get_flows();
118  const auto& model = this->get_model();
119  auto& result = this->get_result();
120  // take the last time point as base result (instead of the initial results), so that we use external changes
121  const size_t last_tp = result.get_num_time_points() - 1;
122  // calculate new time points
123  for (Eigen::Index i = result.get_num_time_points(); i < flows.get_num_time_points(); i++) {
124  result.add_time_point(flows.get_time(i));
125  model.get_derivatives(flows.get_value(i) - flows.get_value(last_tp), result.get_value(i));
126  result.get_value(i) += result.get_value(last_tp);
127  }
128  }
